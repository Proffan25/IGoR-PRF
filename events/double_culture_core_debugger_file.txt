province_event = {
	id = 78700943
	title = "Platinean Core Debugging"
	desc = "If there is an awkward overlapping culture core, this event is intended to clean it up"
	
	trigger = {
		owner = {
			OR = {
				tag = PRG
				tag = ARG
				tag = URU
				tag = CHL
			}
			NOT = {
				war = yes
			}
		}
		OR = {
			AND = {
				is_core = ARG
				is_core = URU
			}
			AND = {
				is_core = ARG
				is_core = CHL
			}
			AND = {
				is_core = ARG
				is_core = PRG
			}
			AND = {
				is_core = URU
				is_core = CHL
			}
			AND = {
				is_core = URU
				is_core = PRG
			}
			AND = {
				is_core = PRG
				is_core = CHL
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "Fix me up!"
		remove_core = PRG
		remove_core = ARG
		remove_core = URU
		remove_core = CHL
		add_core = THIS
	}
}
